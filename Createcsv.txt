Sub SaveMergedDataAsCSV()
    Dim ws As Worksheet
    Dim csvFilePath As String
    
    ' Set the worksheet containing the merged data
    Set ws = ThisWorkbook.Sheets("Sheet1") ' Change "MergedData" to your sheet name
    
    ' Set the path and name of the CSV file
    csvFilePath = "C:\Users\bmanj\Desktop\merged.csv" ' Change the path to your desired location
    
    ' Save the worksheet as CSV
    ws.Copy
    With ActiveWorkbook
        .SaveAs Filename:=csvFilePath, FileFormat:=xlCSV
        .Close False
    End With
    
    MsgBox "CSV file has been saved to " & csvFilePath

End Sub
